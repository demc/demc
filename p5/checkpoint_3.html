// Negative means ball is not in motion
ballX ← -1
ballY ← -1

PROCEDURE setup
   createCanvas 480, 480


PROCEDURE draw
   background 122
   xMidpoint ← width / 2
   yMidpoint ← height / 2

  // Draw the cursor
   fill 255
   ellipse mouseX, mouseY, 25, 25

   IF mouseIsPressed = true
     ballX ← mouseX
ballY ← mouseY

   isBallFlying ← ballX >= 0 AND ballY >= 0
   IF isBallFlying = true
     IF ballY < yMidpoint
     ballY ← ballY - 9
ELSE
     ballY ← ballY + 9
     IF ballX < xMidpoint
     ballX ← ballX - 9
ELSE
     ballX ← ballX + 9
  
  // Draw the ball
   fill 255, 255, 0
   ellipse ballX, ballY, 25, 25


   isBallOutOfBounds ← ballY < 0 OR ballY> height
   IF isBallOutOfBounds = true
   ballX ← -1
   ballY ← -1
