ballX ← -1
ballY ← -1
speedX ← 0
speedY ← 0

PROCEDURE setup
   createCanvas 400, 400


PROCEDURE draw
   background 255

stroke 0
line width / 2, 0, width / 2, height
line 0, height / 2, width, height / 2

noStroke
fill 255, 0, 0
ellipse ballX, ballY, 20, 20
ballX ← ballX + speedX
ballY ← ballY + speedY

fill 0, 0, 255
ellipse mouseX, mouseY, 20, 20

IF mouseIsPressed = true
   ballX ← mouseX
ballY ← mouseY

IF mouseY < height / 2
   // TOP HALF
speedY ← -5

ELSE
   // BOTTOM HALF
speedY ← 5

IF mouseX < width / 2
   // LEFT HALF
speedX ← -5

ELSE
   // RIGHT HALF
speedX ← 5
